public class Maths{
   public virtual int Add( int a , int b ) ;
}
public class ChildMaths : Maths{
   public override int Add( int a , int b ) {
       while ( b != 0 ) {
         int XorResult = a ^ b ;
         int Carry = ( a & b ) << 1 ;
         a = XorResult ;
         b = Carry ;
       }
     return a ;
   }
}
